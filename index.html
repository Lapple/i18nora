public function safeUp()
{
    $id = $this->_i18n('', '');
    $this->_i18n_translate($id, '');
}

public function safeDown()
{
    $this->_i18n_delete('', '');
}

private function _i18n( $category, $key )
{
    $sql = "INSERT INTO `` (`category`, `status`, `key`)
            VALUES ('{$category}', 1, '{$key}');";
    Yii::app()->db->createCommand($sql)->execute();
    return Yii::app()->db->getLastInsertID();
}

private function _i18n_translate( $id, $text )
{
    $sql = "INSERT INTO `` (`i18n_id`, `language`, `text`)
            VALUES({$id}, '', '{$text}');";
    Yii::app()->db->createCommand($sql)->execute();
}

private function _i18n_delete( $category, $key )
{
    $sql = "SELECT `id` FROM ``
            WHERE `category`='{$category}' AND `key`='{$key}' LIMIT 1";

    $id = Yii::app()->db->createCommand($sql)->queryScalar();

    $sql = "DELETE FROM `` WHERE `id`={$id}";
    Yii::app()->db->createCommand($sql)->execute();

    $sql = "DELETE FROM `` WHERE `i18n_id`={$id}";
    Yii::app()->db->createCommand($sql)->execute();
}
